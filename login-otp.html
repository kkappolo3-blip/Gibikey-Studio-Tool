function handleLogin() { const memberId = document.getElementById('nameLogin').value; if (!memberId) { showAlert('Pilih nama Anda terlebih dahulu!', 'error'); return; } // CEK APAKAH MEMBER DI-BAN const banned = JSON.parse(localStorage.getItem('bannedMembers')) || []; const isBanned = banned.find(b => b.memberId === memberId); if (isBanned) { const now = new Date().getTime(); const sisaMs = isBanned.unbanTime - now; if (sisaMs > 0) { const sisaJam = Math.ceil(sisaMs / (1000 * 60 * 60)); showAlert(`❌ Akun Anda sedang DI-BAN! Alasan: ${isBanned.reason}\n\nSisa waktu ban: ${sisaJam} jam`, 'error'); return; } } const members = JSON.parse(localStorage.getItem('approvedMembers')) || []; const member = members.find(m => m.id === memberId); if (!member) { showAlert('Member tidak ditemukan!', 'error'); return; } sessionStorage.setItem('userSession', JSON.stringify({ id: member.id, nama: member.nama, isAdmin: false, loginTime: new Date().toLocaleString('id-ID') })); showAlert(`✓ Selamat datang ${member.nama}!`, 'success'); setTimeout(() => { window.location.href = 'index.html'; }, 1500); }
